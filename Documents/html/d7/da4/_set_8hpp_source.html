291 // Insert node.
292 Node* insert(Node* pos, const T& element)
293 {
294 // TODO: AVL
295 if (pos == nullptr)

297 pos = new Node(element);
298 size_++;
299 }

303 {

305 }

315 // Remove node.
316 Node* remove(Node* pos, const T& element)
317 {
318 // TODO: AVL
319 if (pos)
320 {
321 if (element < pos->data_)
322 {
323 pos->link_left(remove(pos->left_, element));
324 }
325 else if (pos->data_ < element)
326 {
327 pos->link_right(remove(pos->right_, element));
328 }
329 else // element == pos->data_
330 {
331 if (pos->left_ && pos->right_) // certainly not the min or max pos
332 {
333 Node* tmp = find_min(pos->right_);
334 pos->data_ = tmp->data_;
335 pos->link_right(remove(pos->right_, tmp->data_));
336 }
337 else // may be the min or max pos
338 {
339 // if it is the min or max pos, mark it
340 if (pos == min_)
341 {
342 min_ = end_;
343 }
344 if (pos == max_) // must `if`, not `else if` cuz pos may be root
345 {
346 max_ = end_;
347 }
348
349 Node* tmp = pos;
350 pos = pos->left_ ? pos->left_ : pos->right_;
351 delete tmp;
352 size_--;
353 }
354 }
355 }
356
357 return pos;

371 // Traverse the subtree rooted at the specified node.
372 template <typename F>
373 static void level_action(Node* node, const F& action)
374 {
375 // level order
376 if (node != nullptr)
377 {
378 Deque<Node*> queue;
379 queue.push_back(node);
380 while (!queue.is_empty())
381 {
382 node = queue.pop_front();
383 action(node->data_);
384 if (node->left_)
385 {
386 queue.push_back(node->left_);
387 }
388 if (node->right_)
389 {
390 queue.push_back(node->right_);
391 }
392 }
