607 Integer operator+(const Integer& rhs) const
608 {
609 // if one of the operands is zero, just return another one
610 if (sign_ == '0' || rhs.sign_ == '0')
611 {
612 return sign_ == '0' ? rhs : *this;
613 }
614
615 // if the operands are of opposite signs, perform subtraction
616 if (sign_ == '+' && rhs.sign_ == '-')
617 {
618 return *this - (-rhs);
619 }
620 else if (sign_ == '-' && rhs.sign_ == '+')
621 {
622 return rhs - (-*this);
623 }
624
625 // the sign of two integers is the same and not zero
626
627 // prepare variables
628 int size = std::max(digits_.size_, rhs.digits_.size_) + 1;
629
630 Integer num1 = *this;
631 num1.add_leading_zeros(size - 1 - num1.digits_.size_);
632
633 Integer num2 = rhs;
634 num2.add_leading_zeros(size - 1 - num2.digits_.size_);
635
636 Integer result;
637 result.sign_ = sign_; // the signs are same
638 result.add_leading_zeros(size - 1); // result initially has a 0
639
640 // simulate the vertical calculation
641 auto& a = num1.digits_;
642 auto& b = num2.digits_;
643 auto& c = result.digits_;
644 for (int i = 0; i < size - 1; i++)
645 {
646 c[i] += a[i] + b[i];
647 c[i + 1] = c[i] / 10;
648 c[i] %= 10;
649 }
650
651 // remove leading zeros and return result
652 return result.remove_leading_zeros();

662 {
663 // if one of the operands is zero
664 if (sign_ == '0' || rhs.sign_ == '0')
665 {
666 return sign_ == '0' ? -rhs : *this;

668

670 if (sign_ == '+' && rhs.sign_ == '-')

770 Integer operator/(const Integer& rhs) const
771 {
772 // if rhs is zero, throw an exception
773 if (rhs.sign_ == '0')
774 {
775 throw std::runtime_error("ERROR: Divide by zero.");
776 }
777
778 // if this is zero, just return zero
779 if (sign_ == '0')
780 {
781 return 0;
782 }
783
784 // the sign of two integers is not zero
785
786 // prepare variables
787 int size = digits_.size_ - rhs.digits_.size_ + 1;
788
789 Integer num1 = (*this).abs();
790
791 Integer tmp; // intermediate variable for rhs * 10^i
792 tmp.sign_ = '+'; // positive
793
794 Integer result;
795 result.sign_ = (sign_ == rhs.sign_ ? '+' : '-'); // the sign is depends on the sign of operands
796 result.add_leading_zeros(size - 1); // result initially has a 0
797
798 // calculation
799 const auto& b = rhs.digits_;
800 auto& c = result.digits_;
801 for (int i = size - 1; i >= 0; i--)
802 {
803 tmp.digits_ = List<signed char>({0}) * i + b; // tmp = rhs * 10^i in O(N)
804
805 while (num1 >= tmp) // <= 9 loops, so O(1)
806 {
807 c[i]++;
808 num1 -= tmp;
809 }
810 }
811
812 // if result is zero, set sign to '0'
813 result.sign_ = (result.digits_.size_ == 1 && result.digits_[0] == 0) ? '0' : result.sign_;
814
815 // remove leading zeros and return
816 return result.remove_leading_zeros();
817 }

837 Integer pow(const Integer& exp, const Integer& mod = 0) const
838 {
839 if (exp.is_negative())
840 {
841 return 0;
842 }
843
844 // fast power algorithm
845
846 Integer num = *this;

848 Integer result = 1; // this**0 == 1
849
850 while (!n.is_zero())
851 {
852 if (n.is_odd())
853 {
854 result = mod.is_zero() ? result * num : (result * num) % mod;
855 }
856 num = mod.is_zero() ? num * num : (num * num) % mod;
857 n /= 2; // integer divide
858 }

887 Integer sqrt() const
888 {
889 if (sign_ == '-')
890 {
891 throw std::runtime_error("ERROR: Cannot compute square root of a negative integer.");
892 }
893
894 if (is_zero())
895 {
896 return 0;
897 }

899 {
900 return 1;
901 }
902
903 // as far as possible to reduce the number of iterations
904 Integer cur_sqrt = *this / 2;
905 Integer pre_sqrt = 2;
906
907 while (cur_sqrt != pre_sqrt)
908 {
909 pre_sqrt = cur_sqrt;
910 cur_sqrt = (cur_sqrt + *this / cur_sqrt) / 2;
911 }
